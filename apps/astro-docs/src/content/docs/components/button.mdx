### Usage
import { Steps } from '@astrojs/starlight/components';

<Steps>

1. Import the component:

   ```ts
        import { Button } from "@realpay-ui/button";
   ```

2. Call the component in your tsx (or jsx) file:
    ```tsx
    export const Demo = () => {
        return (
            <Button>Example button</Button>
        )
    }
    ``` 
</Steps>
